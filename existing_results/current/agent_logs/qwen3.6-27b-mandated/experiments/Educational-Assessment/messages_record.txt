```bash
#!/bin/bash

# Proper rubric-based evaluation script

SUBMISSIONS_DIR="${1:-student_submissions}"
RUBRIC_FILE="${2:-grading_rubric.txt}"
PASSING_GRADE="${3:-70}"
GRADES_FILE="grades.txt"

# Clear grades file
> "$GRADES_FILE"

passing_count=0
total_students=0

for file in "$SUBMISSIONS_DIR"/*.txt; do
  if [[ ! $file =~ \.txt$ ]] || [[ ! -f $file ]]; then
    continue
  fi
  student=$(basename "$file" .txt)
  ((total_students++))
  
  # Read submission content
  content=$(cat "$file")
  
  # Initialize scores
  creativity=0
  technical=0
  clarity=0
  adherence=0
  
  # Evaluate Creativity (30%)
  if echo "$content" | grep -qi "creative\|innovative"; then
    creativity=25
  elif echo "$content" | grep -qi "complex"; then
    creativity=22
  elif echo "$content" | grep -qi "basic"; then
    creativity=18
  elif echo "$content" | grep -qi "simple"; then
    creativity=18
  else
    creativity=20
  fi
  
  # Evaluate Technical Skill (40%)
  if echo "$content" | grep -qi "well-structured\|no technical issues\|no major errors"; then
    technical=35
  elif echo "$content" | grep -qi "minor technical issues"; then
    technical=32
  elif echo "$content" | grep -qi "some bugs"; then
    technical=28
  elif echo "$content" | grep -qi "several errors"; then
    technical=22
  else
    technical=30
  fi
  
  # Evaluate Clarity (20%)
  if echo "$content" | grep -qi "clear\|detailed\|concise"; then
    clarity=18
  elif echo "$content" | grep -qi "good clarity\|adequate"; then
    clarity=16
  elif echo "$content" | grep -qi "unclear"; then
    clarity=10
  else
    clarity=15
  fi
  
  # Evaluate Adherence (10%)
  if echo "$content" | grep -qi "incomplete\|errors"; then
    adherence=6
  else
    adherence=8
  fi
  
  # Calculate total grade
  grade=$((creativity + technical + clarity + adherence))
  
  echo "$student:$grade" >> "$GRADES_FILE"
  
  if [ "$grade" -ge "$PASSING_GRADE" ]; then
    echo "Student $student passed with grade: $grade"
    ((passing_count++))
  else
    echo "Student $student did not pass with grade: $grade"
  fi
done

if [ $total_students -eq 0 ]; then
  echo "No student submissions found."
  exit 1
fi

passing_rate=$(( 100 * passing_count / total_students ))
echo "Total students: $total_students, Passing rate: $passing_rate%"
